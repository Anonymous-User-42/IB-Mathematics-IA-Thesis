

\section{\textit{The Fourier Series}}

	\textit{The Fourier Series is a infinite summation representation of sinusoidal functions of a periodic function. Mathematically,}
		
		$$f(x) = \sum_{n = 0}^{\infty}\left(a_n\cdot\cos\left(nx\right) + b_n\cdot\sin\left(nx\right)\right)$$

	\textit{Alternatively,}

		$$f(x) = \sum_{n = 0}^{\infty}a_n\cdot\cos\left(nx\right) + \sum_{n = 0}^{\infty}b_n\cdot\sin\left(nx\right)$$

	\textit{Or,}
	
		$$f(x) = a_0 + \sum_{n = 1}^{\infty}a_n\cdot\cos\left(nx\right) + \sum_{n = 1}^{\infty}b_n\cdot\sin\left(nx\right)$$

	\textit{Where,}
	
		$$a_n = \frac{2}{P}\cdot\int_P{f(x)\cdot\cos\left(\frac{2\pi}{P}\cdot nx\right)dx$$

		$$b_n = \frac{2}{P}\cdot\int_P{f(x)\cdot\sin\left(\frac{2\pi}{P}\cdot nx\right)dx}$$

	\textit{With exceptions to the coefficient when $n = 0$, also where $f(x)$ is a periodic function.}

	\textit{Usually of the form,}
	
		$$a_n = \frac{1}{\pi}\cdot\int_{-\pi}^{+\pi}{f(x)\cdot\cos\left(nx\right)dx$$
		
		$$b_n = \frac{1}{\pi}\cdot\int_{-\pi}^{+\pi}{f(x)\cdot\cos\left(nx\right)dx$$

	\textit{By utilizing Euler's formula, we can convert the series from the real from to complex-exponential form. Therefore we have,}

		$$f(x) = \sum_{n = -\infty}^{\infty}c_n\cdot e^{inx}$$

	\textit{Where,}
	
		$$c_n = \frac{1}{2\pi}\cdot\int_L f(x)\cdot e^{-inx}dx$$
